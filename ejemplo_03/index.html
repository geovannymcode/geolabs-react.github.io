    üìù Construyendo TaskForm.jsx

    Versi√≥n completa comentada

    import { useState } from 'react'

function TaskForm({ onAdd }) {
  // 1. ESTADO LOCAL: Solo para el input
  const [titulo, setTitulo] = useState("")

  // 2. MANEJADOR DE ENV√çO
  const handleSubmit = (e) => {
    // Prevenir recarga de p√°gina
    e.preventDefault()

    // Validaci√≥n: eliminar espacios y verificar
    const valor = titulo.trim()
    if (valor === "") return  // No agregar tareas vac√≠as

    // Comunicar al padre
    onAdd(valor)

    // Limpiar formulario
    setTitulo("")
  }

  // 3. RENDERIZADO
  return (
    <form onSubmit={handleSubmit} style={{ marginBottom: "1rem" }}>
      <input
        type="text"
        value={titulo}                    // Controlado por React
        onChange={(e) => setTitulo(e.target.value)}
        placeholder="Nueva tarea"
        style={{ padding: "0.5rem", width: "80%" }}
      />
      <button 
        type="submit" 
        style={{ padding: "0.5rem" }}
      >
        Agregar
      </button>
    </form>
  )
}

export default TaskForm
