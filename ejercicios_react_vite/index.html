const cache = {
  posts: null,           // Almacenar√° todos los posts
  postDetails: {},       // Almacenar√° posts individuales por ID
  timestamp: null,       // Cu√°ndo se actualiz√≥ por √∫ltima vez la cach√©
  // Cach√© v√°lido por 5 minutos (300,000 ms)
  isValid: () => cache.timestamp && (Date.now() - cache.timestamp < 5 * 60 * 1000)

// Funci√≥n para obtener todos los posts
export async function getPosts() {
  // Si tenemos datos en cach√© v√°lidos, los devolvemos
  if (cache.posts && cache.isValid()) {
    console.log('Usando posts desde cach√©');
    return cache.posts;
  }

  // Si no hay cach√© o expir√≥, hacemos la petici√≥n
  try {
    const response = await fetch('https://jsonplaceholder.typicode.com/posts');

    if (!response.ok) {
      throw new Error(`Error HTTP: ${response.status}`);
    }

    const data = await response.json();

    // Actualizamos la cach√©
    cache.posts = data;
    cache.timestamp = Date.now();

    return data;
  } catch (error) {
    console.error("Error en getPosts:", error);
    throw error;
  }
}

// Funci√≥n para obtener un post por su ID
export async function getPostById(id) {
  // Si tenemos el post espec√≠fico en cach√© y es v√°lido, lo devolvemos
  if (cache.postDetails[id] && cache.isValid()) {
    console.log(`Usando post ${id} desde cach√©`);
    return cache.postDetails[id];
  }

  try {
    const response = await fetch(`https://jsonplaceholder.typicode.com/posts/${id}`);

    if (!response.ok) {
      throw new Error(`Error HTTP: ${response.status}`);
    }

    const data = await response.json();

    // Actualizamos la cach√©
    cache.postDetails[id] = data;
    cache.timestamp = Date.now();

    return data;
  } catch (error) {
    console.error(`Error en getPostById(${id}):`, error);
    throw error;
  }
}
